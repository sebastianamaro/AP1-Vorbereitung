Ihre View:

CREATE View K_A (Kategoriebezeichnung,Anzahl) AS

SELECT K.Bezeichnung, COUNT(R.Bezeichnung) FROM Ressource R
JOIN Kategorie K ON R.Kategorie_ID = K.kategorie_ID
